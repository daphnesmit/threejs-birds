{# An example button macro. Using macro's for buttons is very useful as it keeps the implementation of a button consistent and easily maintainable as you only have a single place where it's defined. Usage: {% include 'elements/button.html' %} {{ button('label here' ) }} {{ button('different label here', 'js-class', 'http://www.tamtam.nl' ) }} #} {% macro button( label, classes= '', url= '#', attributes = '' ) %} {{ label }} {% endmacro %}
